    <script>
        let barcodeDb = {}; 
        let isNavigating = false; 

        const statusBox = document.getElementById('statusBox');
        const statusText = document.getElementById('statusText');

        function setStatus(message, type) {
            statusText.innerHTML = message;
            statusText.className = '';
            if (type) statusText.classList.add(type);
        }

        // 1. 엑셀 파일(.xlsx) 로드 함수
        async function loadExcel() {
            try {
                // 깃허브의 Excel 폴더 안에 있는 barcode_mapping.xlsx 파일을 읽어옵니다.
                const response = await fetch('Excel/barcode_mapping.xlsx');
                if (!response.ok) throw new Error("엑셀 파일을 찾을 수 없습니다.");
                
                const arrayBuffer = await response.arrayBuffer();
                const workbook = XLSX.read(arrayBuffer, { type: 'array' });
                
                // 첫 번째 시트의 데이터를 가져옵니다.
                const firstSheetName = workbook.SheetNames[0];
                const worksheet = workbook.Sheets[firstSheetName];
                
                // 데이터를 2차원 배열 형태로 변환합니다.
                const data = XLSX.utils.sheet_to_json(worksheet, { header: 1 });
                
                // 첫 번째 줄(헤더: Barcode, URL 등)은 건너뛰고 2번째 줄(인덱스 1)부터 읽습니다.
                for (let i = 1; i < data.length; i++) {
                    const row = data[i];
                    if (row.length >= 2) {
                        const barcode = String(row[0]).trim(); // A열 (바코드)
                        const url = String(row[1]).trim();     // B열 (URL)
                        
                        if (barcode && url) {
                            if (!barcodeDb[barcode]) barcodeDb[barcode] = [];
                            barcodeDb[barcode].push(url);
                        }
                    }
                }
                
                console.log("✅ 엑셀 데이터 로드 완료:", barcodeDb);
                setStatus('✅ 카메라 권한을 허용하고 바코드를 스캔하세요.', 'status-success');
                startScanner();
                
            } catch (error) {
                console.error(error);
                setStatus(`❌ 데이터 로드 에러: ${error.message}<br>Excel/barcode_mapping.xlsx 파일이 올바른 위치에 있는지 확인하세요.`, 'status-error');
            }
        }

        // 2. 바코드 스캔 로직
        function onScanSuccess(decodedText, decodedResult) {
            if (isNavigating) return;

            const urls = barcodeDb[decodedText];
            
            if (urls && urls.length > 0) {
                isNavigating = true;
                html5QrcodeScanner.pause(); 
                
                const targetUrl = urls[0];
                setStatus(`🎉 스캔 성공! [ ${decodedText} ]<br>해당 링크로 이동합니다...`, 'status-success');
                
                setTimeout(() => {
                    window.location.href = targetUrl;
                }, 1000);
            } else {
                setStatus(`❌ 엑셀에 등록되지 않은 바코드입니다: ${decodedText}`, 'status-error');
            }
        }

        let html5QrcodeScanner;

        // 3. 스캐너 실행 함수
        function startScanner() {
            html5QrcodeScanner = new Html5QrcodeScanner(
                "reader", 
                { 
                    fps: 10, 
                    qrbox: { width: 250, height: 150 },
                    aspectRatio: 1.0,
                    showTorchButtonIfSupported: true 
                }, 
                false
            );
            html5QrcodeScanner.render(onScanSuccess);
        }

        // 앱 시작 시 엑셀 로드 실행
        window.onload = loadExcel;
    </script>
